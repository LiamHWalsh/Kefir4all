#mkdir /data/Food/analysis/R6564_NGS/kefir4all/03_MAG_classification/03_dram/

find /data/Food/analysis/R6564_NGS/kefir4all/06_strain_profiling/06_snippy/06_snippy_reference_genome -type f -name "*.fna" -exec mv {} /data/Food/analysis/R6564_NGS/kefir4all/03_MAG_classification/03_dram/NCBI_genomes/ \;

source activate DRAM

module unload dram/1.2
