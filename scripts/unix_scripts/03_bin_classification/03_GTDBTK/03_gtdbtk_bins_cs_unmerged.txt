#!/bin/sh
#SBATCH --error 03_err_gtdbtk_bins_cs_unmerged
#SBATCH --output 03_out_gtdbtk_bins_cs_unmerged
#SBATCH --job-name 03_gtdbtk_bins_cs_unmerged
#SBATCH -p Priority,Background,GPU
#SBATCH -N 1
#SBATCH --cpus-per-task=40   
#SBATCH --mail-type=BEGIN,END,FAIL
#SBATCH --mail-user=liam.walsh@teagasc.ie


mkdir /data/Food/analysis/R6564_NGS/kefir4all/03_MAG_classification/03_gtdbtk/gtdbtk_bins_cs_unmerged

module load  gtdb-tk/1.3.0

 
gtdbtk identify --genome_dir /data/Food/analysis/R6564_NGS/kefir4all/02_assembly/02_metawrap/02_metawrap_unmerged/finished_cs_bins/ --out_dir /data/Food/analysis/R6564_NGS/kefir4all/03_MAG_classification/03_gtdbtk/gtdbtk_bins_cs_unmerged/identify -x .fa --cpus $SLURM_CPUS_PER_TASK

gtdbtk align --identify_dir /data/Food/analysis/R6564_NGS/kefir4all/03_MAG_classification/03_gtdbtk/gtdbtk_bins_cs_unmerged/identify --out_dir /data/Food/analysis/R6564_NGS/kefir4all/03_MAG_classification/03_gtdbtk/gtdbtk_bins_cs_unmerged/align --cpus $SLURM_CPUS_PER_TASK

gtdbtk classify --genome_dir /data/Food/analysis/R6564_NGS/kefir4all/02_assembly/02_metawrap/02_metawrap_unmerged/finished_cs_bins/ --align_dir /data/Food/analysis/R6564_NGS/kefir4all/03_MAG_classification/03_gtdbtk/gtdbtk_bins_cs_unmerged/align --out_dir /data/Food/analysis/R6564_NGS/kefir4all/03_MAG_classification/03_gtdbtk/gtdbtk_bins_cs_unmerged/align/classify -x .fa --cpus $SLURM_CPUS_PER_TASK

module unload  gtdb-tk/1.3.0
