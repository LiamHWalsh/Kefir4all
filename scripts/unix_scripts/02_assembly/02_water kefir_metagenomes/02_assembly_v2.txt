#!/bin/sh
#SBATCH --error 02err_assembly_sWK_v2_merged
#SBATCH --output 02out_assembly_sWK_v2_merged
#SBATCH --job-name 02_assembly_sWK_v2_merged
#SBATCH -p Priority,Background,GPU
#SBATCH -N 1
#SBATCH --cpus-per-task=35
#SBATCH --mail-type=BEGIN,END,FAIL
#SBATCH --mail-user=liam.walsh@teagasc.ie

#rm -r /data/Food/analysis/R6564_NGS/kefir4all/02_assembly/02_metawrap
#mkdir /data/Food/analysis/R6564_NGS/kefir4all/02_assembly/02_metawrap


for base_name in $(find /data/Food/analysis/R6564_NGS/kefir4all/01qc/01bowtie -type f \( -name "WK*" -and -name "*.sam" \) | cut -d/ -f9 | cut -d- -f1 | sort -u )
do
#cd /data/Food/analysis/R6564_NGS/kefir4all/01qc/01fastq_files/merged/interleaved/
#gunzip /data/Food/analysis/R6564_NGS/kefir4all/01qc/01fastq_files/merged/interleaved/"$base_name"_interleaved.fastq.gz


module load metawrap/1.3.2
#module load metawrap/1.3.1
module load checkm/1.0.18
module load bowtie2/2.3.4
module load samtools/1.10
module load metabat2/2.12.1
module load maxbin2/2.2.7
module load concoct/1.1.0
source activate concoct_env

#mkdir /data/Food/analysis/R6564_NGS/kefir4all/02_assembly/02_metawrap/"$base_name"_metawrap_bins


metaspades.py -1 /data/Food/analysis/R6564_NGS/kefir4all/01qc/01fastq_files/merged/"$base_name"_merged_R1.fastq.gz -2 /data/Food/analysis/R6564_NGS/kefir4all/01qc/01fastq_files/merged/"$base_name"_merged_R2.fastq.gz -t \
$SLURM_CPUS_PER_TASK -o /data/Food/analysis/R6564_NGS/kefir4all/02_assembly/02_metaspades/02_metaspades_merged/"$base_name"_metaspades


metawrap binning -o /data/Food/analysis/R6564_NGS/kefir4all/02_assembly/02_metawrap/02_metawrap_merged/"$base_name"_metawrap_bins -t $SLURM_CPUS_PER_TASK -a /data/Food/analysis/R6564_NGS/kefir4all/02_assembly/02_metaspa\
des/02_metaspades_merged/"$base_name"_metaspades/contigs.fasta --metabat2 --maxbin2 --concoct --interleaved /data/Food/analysis/R6564_NGS/kefir4all/01qc/01fastq_files/merged/interleaved/"$base_name"_interleaved.fastq

metawrap bin_refinement -o /data/Food/analysis/R6564_NGS/kefir4all/02_assembly/02_metawrap/02_metawrap_merged/"$base_name"_metawrap_refinement -t $SLURM_CPUS_PER_TASK -A /data/Food/analysis/R6564_NGS/kefir4all/02_assemb\
ly/02_metawrap/02_metawrap_merged/"$base_name"_metawrap_bins/metabat2_bins/ -B /data/Food/analysis/R6564_NGS/kefir4all/02_assembly/02_metawrap/02_metawrap_merged/"$base_name"_metawrap_bins/maxbin2_bins/ -C /data/Food/an\
alysis/R6564_NGS/kefir4all/02_assembly/02_metawrap/02_metawrap_merged/"$base_name"_metawrap_bins/concoct_bins/ -c 50 -x 5

metawrap reassemble_bins -o /data/Food/analysis/R6564_NGS/kefir4all/02_assembly/02_metawrap/02_metawrap_merged/"$base_name"_metawrap_bin_reassembly -1 /data/Food/analysis/R6564_NGS/kefir4all/01qc/01fastq_files/merged/"$\
base_name"_merged_R1.fastq.gz -2 /data/Food/analysis/R6564_NGS/kefir4all/01qc/01fastq_files/merged/"$base_name"_merged_R2.fastq.gz -t $SLURM_CPUS_PER_TASK -m 800 -c 50 -x 5 -b /data/Food/analysis/R6564_NGS/kefir4all/02_\
assembly/02_metawrap/02_metawrap_merged/"$base_name"_metawrap_refinement/metawrap_50_5_bins
module unload metawrap/1.3.2
#module unload metawrap/1.3.1
module unload checkm/1.0.18
module unload bowtie2/2.3.4
module unload samtools/1.10
module unload metabat2/2.12.1
module unload maxbin2/2.2.7
module unload concoct/1.1.0
conda deactivate


done
